# docker-compose.yaml

### Разберем миграции 
```yaml
todoapp-postgres-migrate:
  image: migrate/migrate:v4.19.1
  volumes: 
    - ${PROJECT_ROOT}/migrations:/migrations
```

Просто создаем контейнер миграций 

# Makefile

### migrate-create
```Makefile
migrate-create:
	@if [ -z "$(seq)" ]; then \
		echo "отсутствует параметр `seq`. Пример: make migrate-create seq=init"; \
		exit 1; \
	fi; \
	docker compose run --rm todoapp-postgres-migrate \
    create \
		-ext sql \
		-dir /migrations \
		-seq "$(seq)"
```

```@if [ -z "$(seq)" ]; then \``` - говорим, что условие выполниться только при не передаче seq  
```echo "отсутствует параметр `seq`. Пример: make migrate-create seq=init"; \``` - сообщенеи ошибки  
```exit 1; \``` - заверщение программы с кодом 1  

В строке: ```docker compose run --rm todoapp-postgres-migrate \```  
```run``` - для не долгоживущих контейнеров  
```--rm``` - удалить контейнер после остановки  
```todoapp-postgres-migrate``` - какой контейнер запускать    

Далее  
```create \``` - явно ей говорим, что надо создать новые файлы миграции
```-ext sql \``` - передаем язык на котором будем писать  
```-dir /migrations \``` - пишем директорию в которой искать/создовать файлы миграции  
```-seq "$(seq)"``` - название версии миграции  
в этой строке:```"$(seq)"``` - говорим подставить то значение, которое мы передодим при вызове таргета (пишем наш таргет и потом: seq={Значение})  


### migrate-up

```Makefile
migrate-action:
	@if [ -z "$(action)" ]; then \
			echo "Отсутствует параметр 'action'.Пример: make migrate-action action=up"; \
			exit 1; \
		fi; \
	docker compose run --rm todoapp-postgres-migrate \
		-path /migrations \
		-database postgres://${POSTGRES_USER}:${POSTGRES_PASSWORD}@todoapp-postgres:5432/${POSTGRES_DB}?sslmode=disable \
		"$(action)"
```
До этого проверки на параметр

Через ```migrate-action``` вызываем это ```docker compose run --rm todoapp-postgres-migrate \```  
и внего передаем путь до ```-path /migrations \```  
и задаем строку подключения к бд в строке ```-database postgres://${POSTGRES_USER}:${POSTGRES_PASSWORD}@todoapp-postgres:5432/${POSTGRES_DB}?sslmode=disable \```  
И передаем параметр ```"$(action)"```

### создание и откат миграций на **нулевую** версию через migrate-action:
```Makefile
migrate-up:
	make migrate-action action=up
migrate-down:
	make migrate-action action=down
```


# Описание схемы данных 
## Всегда начинаем с повышения миграции схемы данных верх 000001_init.up.sql
```sql
CREATE SCHEMA todoapp;

CREATE TABLE todoapp.users(
  id           SERIAL                PRIMARY KEY,
  version      BIGINT       NOT NULL DEFAULT 1,
  full_name    VARCHAR(100) NOT NULL CHECK  (char_length(full_name) BETWEEN 3 AND 100),
  phone_number VARCHAR(15)           CHECK  (
    phone_number ~ '^\+[0-9]+$'
    AND
    char_length(phone_number)  BETWEEN 10 AND 15
  )
);

CREATE TABLE todoapp.tasks (
  id             SERIAL                PRIMARY KEY,
  version        BIGINT       NOT NULL DEFAULT 1,
  title          VARCHAR(100) NOT NULL CHECK(char_length(title) BETWEEN 1 AND 100),
  description    VARCHAR(1000)         CHECK(char_length(description) BETWEEN 1 AND 100),
  completed      BOOLEAN      NOT NULL,
  created_at     TIMESTAMPTZ  NOT NULL,
  completed_at   TIMESTAMPTZ,

  CHECK (
    (completed=FALSE AND completed_at IS NULL)
    OR
    (completed=TRUE AND completed_at IS NOT NULL AND completed_at >= created_at)
  ),
  
  author_user_id INTEGER NOT NULL REFERENCES todoapp.users(id)
);

```

```CREATE SCHEMA todoapp;``` - созданеи схемы в которой и будут наши таблици

### Таблица USERS
```sql
CREATE TABLE todoapp.users(
  id           SERIAL                PRIMARY KEY,
  version      BIGINT       NOT NULL DEFAULT 1,
  full_name    VARCHAR(100) NOT NULL CHECK  (char_length(full_name) BETWEEN 3 AND 100),
  phone_number VARCHAR(15)           CHECK  (
    phone_number ~ '^\+[0-9]+$'
    AND
    char_length(phone_number)  BETWEEN 10 AND 15
  )
);
```

```CREATE TABLE todoapp.users(``` - создание таблицы в схеме todoapp с названием users  
```id SERIAL PRIMARY KEY,``` - Создаем колонку id (первичный ключ)  
```version BIGINT NOT NULL DEFAULT 1,``` - колонка version (не пустая и стандартное значение ***1*** )  
```full_name VARCHAR(100) NOT NULL CHECK (char_length(full_name) BETWEEN 3 AND 100),``` - имя пользователя (не пустое и проверка на длинну от 3 до 100 символов)  
```phone_number VARCHAR(15) CHECK  (``` - номер телефона и начало проверки  

```phone_number ~ '^\+[0-9]+$'``` - регулярное выражение:  
- \^\\+    
  * ^ - говорит что первый символ должен...
  * \ - говорит что след сивол берем как символ, а не как регулярное выражение
  * \+ - берем именно сивол "+" в начало строки     
- [0-9] - символы только от 0 до 9 
- +$ - символы могут повторяться



```AND``` - и  
```char_length(phone_number)  BETWEEN 10 AND 15``` - проверка на длинну от 10 до 15 символов

Пофакту тоже самое как и выше только:  
```sql
CHECK (
    (completed=FALSE AND completed_at IS NULL)
    OR
    (completed=TRUE AND completed_at IS NOT NULL AND completed_at >= created_at)
  ),
```
^ проверка на то что если выполнена задача, то даат начала меньше даты конца  

```author_user_id INTEGER NOT NULL REFERENCES todoapp.users(id)``` - подключаем к полю ***author_user_id*** таблици ***tasks*** к схеме ***todoapp*** таблици ***users***поля ***id***  

## Понижение миграции 000001_init.down.sql
Выполниться в обратной последовательности от повышения миграции

```sql
DROP TABLE todoapp.tasks;
DROP TABLE todoapp.users;
DROP SCHEMA todoapp;
```

```DROP TABLE todoapp.tasks;``` - удаляем табл задачи  

```DROP TABLE todoapp.users;``` - удаляем табл пользователи  

```DROP SCHEMA todoapp;```      - удаляем схему
